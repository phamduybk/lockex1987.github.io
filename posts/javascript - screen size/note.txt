	function getScreenWidthType() {
		var checkWidth = $(window).width();

		if (checkWidth < 480) {
			return 'phone';
		} else if (checkWidth < 768) {
			return 'smallTablet';
		} else if (checkWidth < 992) {
			return 'tablet';
		} else if (checkWidth < 1200) {
			return 'desktop';
		} else  {
			return 'wideDesktop';
		}
	}
