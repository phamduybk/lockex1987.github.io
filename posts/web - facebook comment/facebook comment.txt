Thêm Facebook comment cho website

	https://developers.facebook.com/apps/
	https://developers.facebook.com/setup/ (tạo app Facebook, public app)
	https://developers.facebook.com/docs/plugins/comments (DOC)
	https://developers.facebook.com/tools/comments (Quản lý)
	

Thêm
<meta property="fb:app_id" content="1762996047326114" />



<div id="fb-root"></div>

<script id="facebook-jssdk" src="https://connect.facebook.net/vi_VN/sdk.js#xfbml=1&version=v2.10&appId=1762996047326114"></script>



<script>

(function() {

// Chú ý data-href đang phân biệt cả giao thức (HTTP và HTTPS)

// Chuyển về HTTPS hết

var href = location.href;

if (!href.startsWith("https")) {

href = "https" + href.substr(4);

}



// Vùng div chứa comment

document.write('<div class="fb-comments" data-href="' + href + '" data-numposts="5" data-order-by="reverse_time"></div>');

})();

</script>
